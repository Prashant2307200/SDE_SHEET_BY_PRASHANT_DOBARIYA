int lengthOfLongestSubstring(string s) {
    int n = s.size();
    if(n == 0 || n == 1)    return n;
    unordered_map<int,int> hsh;
    int maxi = 1 ,j = 0;
    for(int i = 0; i < n; ++i){
        if(hsh.find(s[i]) != hsh.end()){
            j = max(j,hsh[s[i]]+1);
        }
        maxi = max(maxi,i-j+1);
        hsh[s[i]] = i;
    }
    return maxi;
}
