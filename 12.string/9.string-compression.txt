int compress(vector<char>& chars) {
    int n = chars.size();
    int ansIdx = 0;
    for(int i = 0; i < n; ++i){
        char ch = chars[i];
        int st = i;
        while(i < n-1 && ch == chars[i+1])  i++;
        int en = i;
        int cnt = en-st+1;
        chars[ansIdx++] = ch;
        if(cnt > 1){
            string count = to_string(cnt);
            for(auto& ch : count)
                chars[ansIdx++] = ch;
        }
    }return ansIdx;
}
