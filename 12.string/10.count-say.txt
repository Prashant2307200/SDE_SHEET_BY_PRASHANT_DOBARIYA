string countAndSay(int n) {
    if(n == 1)
        return "1";
    string say = countAndSay(n-1);
    int sz = say.size();
    string result = "";
    for(int i = 0; i < sz; ++i) {
        int count = 1;
        char ch = say[i];
        while(i < sz-1 && say[i] == say[i+1]) {
            count++;
            i++;
        }result += to_string(count) + string(1, say[i]);
    }return result;
}
