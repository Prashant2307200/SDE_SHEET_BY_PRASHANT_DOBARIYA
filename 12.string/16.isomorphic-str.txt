bool isIsomorphic(string s, string t) {
    int n = s.size();
    int m = t.size();
    unordered_map<char,char> hsh1,hsh2;
    if(n != m){
        return 0;
    }for(int i = 0; i < n; ++i){
        if(hsh1[s[i]] && hsh1[s[i]] != t[i]) return 0;
        if(hsh2[t[i]] && hsh2[t[i]] != s[i]) return 0;
        hsh1[s[i]] = t[i];
        hsh2[t[i]] = s[i];
    }return 1;
}
