int maxSubArray(vector<int>& nums) {
    int maxSum = nums[0] ,currSum = 0 ,n = nums.size();
    for(int i = 0; i < n; ++i){
        currSum += nums[i];
        if(currSum < 0)
            currSum = 0;
        maxSum = max(maxSum ,currSum);
    }
    return maxSum == 0? *max_element(nums.begin(),nums.end()) : maxSum;
}
