void f(int num,vector<string>& ans,int i,string binary){
    if(i == num){
        ans.push_back(binary);
        return;
    }
    if(i != 0 && binary[i-1] == '1'){
        f(num,ans,i+1,binary+"0");
    }
    else{
        f(num,ans,i+1,binary+"0");
        f(num,ans,i+1,binary+"1");
    }
}
vector<string> generateBinaryStrings(int num){
    vector<string> ans;
    f(num,ans,0,"");
    return ans;
}
