int firstOcc(int arr[],int n,int i,int key){
    if(i == n)  return -1;
    if(arr[i] == key)   return i;
    return firstOcc(arr,n,i+1,key);
}

int lastOcc(int arr[],int n,int i,int key){
    if(i == n)  return -1;
    int rop = lastOcc(arr,n,i+1,key);
    if(rop != -1)   return rop;
    if(arr[i] == key)   return i;
    return -1;
}
