void insertAtBottom(stack<int>& s,int insertElm){
    if(s.empty()){
        s.push(insertElm);
        return;
    }int elm = s.top();
    s.pop();
    insertAtBottom(s,insertElm);
    s.push(elm);
}

void rev(stack<int>& s){
    if(s.empty()){
        return;
    }int elm = s.top();
    s.pop();
    rev(s);
    insertAtBottom(s,elm);
}
